提交&签名
# 提交 & 签名
$ git commit -S -m "..."

# 全局配置
$ git config --global commit.gpgsign true

# 查看
$ git log --show-signature

# 导入Github签名
$ curl https://github.com/web-flow.gpg | gpg --import
# 信任签名（用自己的密钥为其签名验证，需要输入密码）
$ gpg --sign-key 4AEE18F83AFDEB23
