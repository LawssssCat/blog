    return 0;
  }

count_primes.rs
extern "C" {
  fn is_prime(n: i32) -> i32;
}

fn main() {
  let n = 50;
  let mut count = 0;

  for i in 2..=n {
    // SAFETY: We're calling a well-behaved external C function
    let result = unsafe { is_prime(i) };
    if result != 0 {
      count += 1;
    }
  }

  println!("Number of prime numbers between 0 and {} is: {}", n, count);
}
