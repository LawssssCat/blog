  private int count = 0;
  public void lock() {

    if (current == owner.get()) {
      count++;
      return;
    }
    while(!owner.compareAndSet(null, current)) {
      // wait ... or else cpu busy
    }

  public void unlock() {
    Thread current = Thread.currentThread();

      }
